        // SDF Function
        function evaluateField(pos, vol, time) {
            const halfSize = wireframeSize / 2;
            const ax = Math.abs(pos.x);
            const ay = Math.abs(pos.y);
            const az = Math.abs(pos.z);
            
            const theoreticalRadius = vol * (halfSize * 1.38);
            const reach = theoreticalRadius / halfSize;
            
            let tension = 0;
            const activationStart = 0.98;
            const activationRange = 0.35;
            
            if (reach > activationStart) {
                let t = (reach - activationStart) / activationRange;
                t = Math.min(1, Math.max(0, t));
                tension = Math.pow(t, 4);
            }
            
            const overflow = Math.max(0, vol - 1.1);
            const tensionDecay = Math.max(0, 1.0 - overflow * 0.6);
            const effectiveTension = tension * tensionDecay;

            const massCorrection = effectiveTension * 0.13;
            const baseRadius = theoreticalRadius * (1.0 - massCorrection);

            let d = baseRadius - pos.length();

            const nx = Math.min(1, ax / halfSize);
            const ny = Math.min(1, ay / halfSize);
            const nz = Math.min(1, az / halfSize);
            const edgeProx = (nx * ny) + (nx * nz) + (ny * nz);
            
            if (effectiveTension > 0) {
                d += edgeProx * 0.14 * effectiveTension;
            }

            const boxDist = Math.max(ax, ay, az) - halfSize;
            
            if (boxDist > 0) {
                const membraneStiffness = Math.max(0.15, 12.0 / (vol * vol * 2.2 + 0.1));
                d -= boxDist * membraneStiffness;
            } else {
                if (effectiveTension > 0.1 && vol < 1.1) {
                    const faceSuck = (1.0 - edgeProx * 0.55) * (1.1 - vol);
                    d -= faceSuck * 0.32 * effectiveTension;
                    
                    if (d < 0.13 && vol > 0.2 && edgeProx < 0.5) {
                        d = Math.max(0.09, d);
                    }
                }
            }

            const noise = Math.sin(pos.x * 3.5 + time) * Math.cos(pos.y * 3.5 + time) * Math.sin(pos.z * 3.5);
            if (d > -0.16 && d < 0.16) {
                d += noise * 0.06;
            }
            
            return d;
        }

        // LEFT SYSTEM (1x1)
        const leftPositions = [new THREE.Vector3(0, 0, 0)];

        // CENTER SYSTEM (32x32)
        const GRID_RES = 32;

        const centerGeo = new THREE.BoxGeometry(voxSize * 0.94, voxSize * 0.94, voxSize * 0.94);
        
        const centerSolidMesh = new THREE.InstancedMesh(centerGeo,
            new THREE.MeshStandardMaterial({ color: 0x00ffff, roughness: 0.1, metalness: 0.2, transparent: true, opacity: 0.5 }),

        centerSolidMesh.frustumCulled = false;

        scene.add(centerSolidMesh);

            new THREE.MeshStandardMaterial({ color: 0x00ff88, roughness: 0.1, metalness: 0.2, transparent: true, opacity: 0.9 }),
            TOTAL_CENTER);
        centerHollowMesh.frustumCulled = false;
        scene.add(centerHollowMesh);

        const centerPositions = [];
        const centerVelocities = new Float32Array(TOTAL_CENTER * 3 * 2);
        const offset = SIM_SIZE / 2;

        for(let x=0; x<GRID_RES; x++) {
            for(let y=0; y<GRID_RES; y++) {
                for(let z=0; z<GRID_RES; z++) {
                    const px = (x / GRID_RES) * SIM_SIZE - offset + voxSize/2;
                    const py = (y / GRID_RES) * SIM_SIZE - offset + voxSize/2;
                    const pz = (z / GRID_RES) * SIM_SIZE - offset + voxSize/2;
                    centerPositions.push(new THREE.Vector3(px, py, pz));
                }
            }
        }

        // RIGHT SYSTEM (10x10)
        const rightRes = 10;
        const rightTotal = rightRes ** 3;
        const rightVoxSize = SIM_SIZE / rightRes;
        const rightGeo = new THREE.BoxGeometry(rightVoxSize * 0.95, rightVoxSize * 0.95, rightVoxSize * 0.95);
        const rightMat = new THREE.MeshStandardMaterial({ color: 0xffcc00, metalness: 0.2, roughness: 0.3 });
        const rightMesh = new THREE.InstancedMesh(rightGeo, rightMat, rightTotal);

        rightMesh.frustumCulled = false;

            for(let y=0; y<rightRes; y++) {
                for(let z=0; z<rightRes; z++) {
                    const px = (x / rightRes) * SIM_SIZE - offset + rightVoxSize/2;
                    const py = (y / rightRes) * SIM_SIZE - offset + rightVoxSize/2;
                    const pz = (z / rightRes) * SIM_SIZE - offset + rightVoxSize/2;
                    rightPositions.push(new THREE.Vector3(px, py, pz));
                }
            }
        }

        // Process voxel with collision
        function processVoxelWithCollision(localPos, velArray, velIdx, mesh, instanceIdx, scale) {
            tempVec.copy(localPos);
            tempVec.x += velArray[velIdx];
            tempVec.y += velArray[velIdx+1];
            tempVec.z += velArray[velIdx+2];

            const worldPos = tempVec.clone().add(mesh.position);
            const posInProbe = worldPos.applyMatrix4(probeInvMatrix);
            
            if (Math.abs(posInProbe.x) < probeHalfSize && 
                Math.abs(posInProbe.y) < probeHalfSize && 
                Math.abs(posInProbe.z) < probeHalfSize) {
                
                const dx = probeHalfSize - Math.abs(posInProbe.x);
                const dy = probeHalfSize - Math.abs(posInProbe.y);
                const dz = probeHalfSize - Math.abs(posInProbe.z);
                
                let pushDir = new THREE.Vector3();
                let pen = 0;
                
                if (dx < dy && dx < dz) {
                    pushDir.set(Math.sign(posInProbe.x), 0, 0); pen = dx;
                } else if (dy < dz) {
                    pushDir.set(0, Math.sign(posInProbe.y), 0); pen = dy;
                } else {
                    pushDir.set(0, 0, Math.sign(posInProbe.z)); pen = dz;
                }
                
                const pushWorld = pushDir.transformDirection(probe.matrixWorld);
                const force = pen * 0.45;
                
                velArray[velIdx] += pushWorld.x * force;
                velArray[velIdx+1] += pushWorld.y * force;
                velArray[velIdx+2] += pushWorld.z * force;
                
                tempVec.addScaledVector(pushWorld, pen);
            }

            velArray[velIdx] *= 0.88;
            velArray[velIdx+1] *= 0.88;
            velArray[velIdx+2] *= 0.88;

            dummy.position.copy(tempVec);
            dummy.scale.setScalar(scale);
            dummy.updateMatrix();
            mesh.setMatrixAt(instanceIdx, dummy.matrix);
        }

        // Update
        function updateSimulation() {
            state.volume += (state.targetVolume - state.volume) * 0.08;
            state.time += 0.015;
            
            probe.updateMatrixWorld();
            probeInvMatrix.copy(probe.matrixWorld).invert();

            const vol = state.volume;
            
            // LEFT (1x1)
            let countLeft = 0;
            const densityLeft = evaluateField(leftPositions[0], vol, state.time);
            if (densityLeft > 0) {
                const s = Math.min(SIM_SIZE * 0.95, vol * 2.6);
                processVoxelWithCollision(leftPositions[0], leftVelocities, 0, leftMesh, countLeft++, s);
            }
            leftMesh.count = countLeft;
            leftMesh.instanceMatrix.needsUpdate = true;

            // CENTER (32x32)
            let countHollow = 0;
            let countSolid = 0;
            const shellThickness = 0.7;
            const resSq = GRID_RES * GRID_RES;

            for (let i = 0; i < TOTAL_CENTER; i++) {
                const ix = Math.floor(i / resSq);
                const rem = i % resSq;
                const iy = Math.floor(rem / GRID_RES);
                const iz = rem % GRID_RES;

                const localPos = centerPositions[i];
                const density = evaluateField(localPos, vol, state.time);
                
                if (density > 0) {
                    if (state.showSolid) {
                        const idxS = i * 3;
                        let scale = 1.0;
                        if (density < 0.2) scale = density * 5.0;
                        scale = Math.max(0.1, Math.min(1, scale));
                        processVoxelWithCollision(localPos, centerVelocities, idxS, centerSolidMesh, countSolid++, scale);
                    }

                    const isGridBorder = (ix === 0 || ix === GRID_RES-1 || 
                                          iy === 0 || iy === GRID_RES-1 || 
                                          iz === 0 || iz === GRID_RES-1);

                    if (density < shellThickness || isGridBorder || density <= 0.14) {
                        const idxH = (TOTAL_CENTER + i) * 3;
                        let scale = 1.0;
                        if (density < 0.2) scale = density * 5.0;
                        scale = Math.max(0.1, Math.min(1, scale));
                        processVoxelWithCollision(localPos, centerVelocities, idxH, centerHollowMesh, countHollow++, scale);
                    }
                }
            }
            
            centerHollowMesh.count = countHollow;
            centerHollowMesh.instanceMatrix.needsUpdate = true;
            if (state.showSolid) {
                centerSolidMesh.count = countSolid;
                centerSolidMesh.instanceMatrix.needsUpdate = true;
            }

            // RIGHT (10x10)
            let countRight = 0;
            for (let i = 0; i < rightTotal; i++) {
                const density = evaluateField(rightPositions[i], vol, state.time);
                if (density > 0) {
                    let scale = 1.0;
                    if (density < 0.18) scale = density / 0.18;
                    scale = Math.max(0.1, Math.min(1, scale));
                    processVoxelWithCollision(rightPositions[i], rightVelocities, i * 3, rightMesh, countRight++, scale);
                }
            }
            rightMesh.count = countRight;
            rightMesh.instanceMatrix.needsUpdate = true;

            document.getElementById('count-left').innerText = countLeft;
            document.getElementById('count-hollow').innerText = countHollow;
            document.getElementById('count-solid').innerText = state.showSolid ? countSolid : 0;
            document.getElementById('count-right').innerText = countRight;
        }
